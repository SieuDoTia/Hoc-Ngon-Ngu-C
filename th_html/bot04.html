บทที่ ๔: โครงสร้างสาขาและวนซ้ำ


เราได้เรียนรู้เกี่ยวกับค่าคงที่ตัวแปรอาร์เรย์แล้ว แต่ที่จะเรียนรู้วิธีที่พวกเขามีการจัดการ โปรแกรมควรเลือกฟังก์ชั่นขึ้นอยู่กับเงื่อนไขการปฏิบัติแล้วซ้ำอีก ภาษา C ที่เรียกว่าแยกทางเรขาคณิตและทำซ้ำจะทำงานนี้

4.1 แยกสาขา

บางครั้งโปรแกรมจะต้องตรวจสอบมูลค่าอันหนึ่งและปฏิบัติตามมูลค่านั้น ตัวอย่างเช่นผู้ใช้พิมพ์อักษรหรือมูลค้าผิดในโปรแกรมและโปรแกรมควรแจ้งให้ผู้ใช้ทราบและเรียกพิมพ์อีกครั้ง หรือคำนวณค่าจ่ายน้ำใช้ถ้าทราบว่า 10 m3 แรกคิด m3 ละ 8 บาท และเกินนั้น m3 ละ 12 บาท การตัดสินใจของโปรแกรมแล้วแต่กรนีเรียกว่าแยกสาขา

4.1.1 if (ถ้า)

   if( <เงื่อนไข> )

      <คำสั่ง>


   if( <เงื่อนไข> )

      <คำสั่ง>

   else

      <คำสั่ง>;


   if( <เงื่อนไข> )

      <คำสั่ง>;

   if else(<เงื่อนไข>)

      <คำสั่ง>;

   ...

   else

      <คำสั่ง>;


สำหรับได้ใช้คำสั่งเพิ่มให้ทุกเงื่อนไขแหนบพวกคำสั่งระหวาง '{' และ '}'

   if(<เงื่อนไข>) {

      <คำสั่ง>

      ...

   }


   if( <เงื่อนไข> ) {

      <คำสั่ง>

      ...

   }

   else {

      <คำสั่ง>

   }


และ

   if( <เงื่อนไข> ) {

      <คำสั่ง>

      ...

   }

   if else( <เงื่อนไข> ) {

      <คำสั่ง>

      ...

   }

   ...

   else {

     <คำสั่ง>

     ...

   }


ถ้าเงื่อนไขเป็นจริงมันมีมูลค่า = 1 เท็จ = 0


ตัวอย่างเช่นคุณอยากเขียนฟังก์ชั่นปรับว่าถ้าจุด (x, y) อยู่ในวงกลมมีรัศมี = 5.0 จากแหล่งแกน




เปลี่ยนเป็นฟังก์ชั่น C:

  1.    /* ในวงกลม */
  2.    unsigned char naiWongKlom( float x, float y) {
  3.     
  4.       float kamLangSong = x*x + y*y;   /* กำลัง ๒ */

  6.       if( kamLangSong <= 25.0f )
  7.          return 1;   /* จริง */
  8.       else
  9.          return 0;   /* เท็จ */
  10.    }

ระมัดระวังกับ ';' ถ้าพิมพ์ผิด

   if( <เงื่อนไข> );     /* <--- มี ';' หลัง if() */

      <คำสั่ง>;         /* ปฏิบัติทุกเวลา          */


ก็เหมื่อน

  if( <เงื่อนไข> )

     ;       /* <--- มี ';' หลัง if() */

  <คำสั่ง>;     /* ปฏิบัติทุกเวลา           */


ก็มีอีกความผิดพลาดคล้ายคลึง:

    if(<เงื่อนไข>)

       <คำสั่ง>;

    else;          /* <--- มี ';' หลังจาก else */

       <คำสั่ง>;     /* ปฏิบัติทุกเวลา */


คุณสามารถใช้ if else ภายใน if else อื่น

   if( <เงื่อนไข 1> ) {

      if( <เงื่อนไข 2> )

          <คำสั่ง>;

       else

          <คำสั่ง>;

    }

    else {

       if( <เงื่อนไข 3> )

          <คำสั่ง>;

       else

          <คำสั่ง>;

    }


ตัวอย่างเช่นพิกัด (x, y) ของจุดหนึ่งและคำนวณมุมจากแกน +x ใช้ฟังก์ชั่นทางคณิตศาสตร์ atan() จากห้องสมุด  math.h จะให้ –π/2 ≤ มุม ≤ π/2 แต่ควรมีเต็มรูปแบบ –π ≤ มุม ≤ π


ฟังก์ชั่น C:

  1.    const float PI 3.1415927f;
  2.    /* คำนวนมุม */
  3.    float khamNuanMum( float x, float y ) {

  5.       float mum = atan( y/x ); /* คำนวณมุม */

  7.       /* รักศาไว้ใน –π ≤ มุม ≤ π */
  8.       if( x < 0.0f ) {
  9.          if( y < 0.0f )
  10.             mum -= PI;
  11.          else
  12.             mum += PI;
  13.       }
  14.       return mum;
  15.    }

ภาษา C มีวิธีการเขียน if() else ในแถว

  <เงื่อนไข> ? <คำสั่งถ้าเป็นจริง> : <คำสั่งถ้าเท็จ>;


สนามครั้งที่ 5-8 ในการทำงาน naiWongKlom() ดังกล่าวข้างต้นจะเขียนใหม่:

   kamLangSong <= 25.0f ? return 1 : return 0; /* ตรวจมูลค่ากำลังสอง */

เรียนรู้เพิ่มเติม

ฉันได้พบมีสองวิธีเพื่อทราบว่าจุดหนึ่งอยู่ข้างในสี่เหลี่มผืนผ้า ขั้นตอนวิธีนี้ได้ใช้ในเกม ตัวอย่างเช่นต้องทราบกระสุนโดนศัตรู


ขั้นตอนวิธีที่มักจะสอนในหนังสือทำเกม:

   /* จุดอยู่ในสี่เหลี่ยม( float จุด_x, จุด_y, ซาย, ขวา, ล้าง, บน ) */

   unsigned char jutYuNaiSiLiam( float jut_x, jut_y, sai, khua, lang, bon ) {


      if( (jut_x > ch_trai) && (jut_x < khua)

                       && (jut_y > lang) && (jut_y < bon) )

         return 1;  /* จุดอยู่ในสี่เหลี่ยม */

      else

         return 0;   /* จุดไม่อยู่ในสี่เหลี่ยม */

   }


วิธีที่สอง:

   /* จุดอยู่ในสี่เหลี่ยม( float จุด_x, จุด_y, ซาย, ขวา, ล้าง, บน ) */

   unsigned char jutYuNaiSiLiam( float jut_x, jut_y, sai, khua, lang, bon ) {


      if( jut_x < sai )

         return 0; /* จุดไม่อยู่ในสี่เหลี่ยม */

      else if( jut_x > khua )

         return 0; /* จุดไม่อยู่ในสี่เหลี่ยม */

      else if( jut_y < lang )

         return 0; /* จุดไม่อยู่ในสี่เหลี่ยม */

      else if (jut_y > bon )

         return 0;  /* จุดไม่อยู่ในสี่เหลี่ยม */

      else

         return 1; /* จุดอยู่ในสี่เหลี่ยม */

   }


นี้สอนเราว่าน่าจะหลายๆ วิธีที่สามารถแก้ปัญหา อันไหนที่ดีที่สุด พูดยาก แล้วแต่กรนีและวิธีใช้มันมีอิทธิพลประสิทธิภาพของขั้นตอนวิธีมาก การใช้งานและจะช่วยส่งผลต่อประสิทธิผลของขั้นตอนวิธี คุณต้องการลองในโปรแกรมและดู!

4.2 วนซ้ำ

ซ้ำเป็นโครงสร้างมีประโยชน์ช่วยทำงานหลายๆ ครั้ง ตัวอย่างนับจำนวนนักเรียนในโรงเรียน การค้นหา  ใช้การแปลงมูลค่าในตัวแปรชุด ฯลฯ ภาษา C มีสามประเภทของวนซ้ำนี้


1. while()

ตัวอย่างสำหรับกวนซ้ำนี้:

   while(<เงื่อนไข>) {

       <คำสั่ง>

       ...

   }


หากเงื่อนไขที่ไม่ถูกต้องก่อนพบวนซ้ำ while() โปรแกรมจะไม่ปฏิบัติคำสั่งระหว่าง '{' และ '}' ตัวอย่างเข่น:

   unsigned char daChaNi = 10; /* ดัชนี = 10 */

   while( daChaNi < 5 ) {   /* daChaNi > 5 จะไม่เข้าวนซ้ำ */

      ...

   }


2. for( ; ; )

ตัวอย่างสำหรับวนซ้ำนี้:

   for(<เริ่มต้นการศึกษา> <เงื่อนไข> <อัพเกรดตัวแปร>) {

       <คำสั่ง>

       ...

   }


จำได้ว่ามันต้องมีเงื่อนไข ทำเช่นนี้ยังถูกต้อง

 • for( ; <เงื่อนไข>; ) {

       <คำสั่ง>

       ...

   }


3. do while()

ตัวอย่างสำหรับกวนซ้ำนี้:

   do {

       <คำสั่ง>

       ...

   } while( <เงื่อนไข> );


มันดูคล้าย while() ที่เราเจอมาก่อนแต่ก็มีจุดหนี่งไม่เหมืื่อน มันตรวจสอบเงื่อนไขหลังจากทำพวกคำสั่งระหว่าง '{' และ '}' ดังนั้นทุกครั้งที่มันจะได้ปฏิบัติคำสั่งอย่างน้อยหนึ่งครั้ง ใช้ในเวลาอยากประกันโปรแกรมจะปฏิบัติวนซ้ำครั้งหนึ่ง


กดใช้วนซ้ำ

• ระดับวนซ้ำ

สามารถว่างวนซ้ำภายในวนซ้ำอื่น ตัวอย่างหนึ่ง:

   unsigned char y = 0;

   while( y < 10 ) {

      unsigned char x = 0;

      for( x=0; x  < 20; x++ ) {

         printf ( "%d; %d\n", x, y);

      }

      y++;

   }


• วนซ้ำไม่บังคับอัพเกรดตัวแปรวิธีหนึ่ง ดูตัวอย่างวนซ้ำ for() คำนวณมูลค่ากำลังสองขนาดเล็ก 70 000:

   unsigned int kamLangSong;       /* กำลังสอง */

   for( kamLangSong = 1; kamLangSong < 70000; kamLangSong *= 2)

      printf ( "%d\n", kamLangSong );


• หลายกรนี while() และ for( ; ; ) ใช้แทนกันได้ ยกตัวอย่างวนซ้ำคำนวณเลขกำลังสองข้างบนสามารถเขียนใช้วนซ้ำ while():

   unsigned int kamLangSong = 1;  /* กำลังสอง */

   while( kamLangSong < 70000 ) {

     kamLangSong *= 2;

      printf ( "%d\n", kamLangSong );

   }


• break (ปล่อย)

คำหลัก break ใช้เพื่อปล่อยจากวนซ้ำ

   while( <เงื่อนไข> ) {

     ...

     if( <เงื่อนไข> )

        break;

     ...

   }


มันปล่อยวนซ้ำระดับหนึ่งเท่านั้นเช่น:

   for( <เริ่มต้นการศึกษา>; <เงื่อนไข>; <อัพเกรดตัวแปร> ) {

      while( <เงื่อนไข> ) {

         ...

         if( <เงื่อนไข> )

            break;

         ...

      }

      /* break จะมาที่นี่ */

   }


• continue (ต่อไป)

หากคุณอยากกระโดดข้ามวนซ้ำแต่ไม่ปล่อยใช้คำหลัก continue

   while( <เงื่อนไข> ) {

      ...

      if( <เงื่อนไข> )

         continue;

      ...

   }


เหมื่อนกับ break มันจะข้ามในวนซ้ำระดับหนึ่ง

   for( <เริ่มต้นการศึกษา>; <เงื่อนไข>; <ทะเลอัพเกรด>) {

      for( <เริ่มต้นการศึกษา>; <เงื่อนไข>; <ตัวแปรอัพเกรด> ) {/* continue จะมาที่นี่ */

         ...

         if( <เงื่อนไข> )

            continue;

         ...

      }

      

   }


หากใช้ do while() เหตุการณ์ข้ามไปถึงเงื่อนไขด้านล่าง

   do {

      ...

      if( <เงื่อนไข> )

            continue;

      ...

   } while( <เงื่อนไข> ) /* continue จะมาที่นี่และดูเงื่อนไข */

บทที่4.2 เปลี่ยนฟังก์ชั่น khamNuanMum() ให้คิดมุมระหว่าง 0 ถึง 2π แล้วก็ฟังก์ชั่นนี้มีความผิดพลาด (ในกรณี x = 0 และ y = 0), แก้ไขมัน


บทที่4.3 เขียนฟังก์แบ่งนักเรียนตามขนาด:

   สุดยอด ≤ 8.0 ≤ 10

   6.5 ≤ ได้ <8.0

   ≤ 5.0 เฉลี่ย <6.5

   3.5 ≤ อ่อนแอ <5.0

   0 ≤ แย้ <3.5


บทที่4.4 หนึ่งวิธีคำนวณรากของจำนวนจริงคือ:

   1. คาดเดามูงค่าแรกใก้ล x0 ≈ √y

   2. หาร y/x0 = z

   3. คำนวณเฉลี่ย x1 = (x + z)/2

   4. ถ้า |z - x1| > ความผิดกลับต้อน 2

  ใช้วิธีนี้เพื่อคำนวณรากมีความผิดน้อยกว่า 0.001 ฟังก์ชั่นควรป้อนกันจากพารามิเตอร์รับจำนวนลบ


บทที่4.5 ทำโปรแกรมรับสามหมายเลขของสมการกำลังสอง:

   ax2 + bx + c = 0

และหามูลค่าให้สองราก x1 และ x2  ค่าของสัมประสิท a, b, c เป็นจำนวนจริง อย่าลืมว่ามันอาจมีบางประสบการณ์จริงเสมือนและซับซ้อน


บทที่4.6 ใน stdlib.h มีฟังก์ชั่นกำเนิดจำนวนสุ่มเรียกว่า rand() เขียนโปรแกรมฝากหืงร้อยจำนวนสุ่มบนหน่าจอ ช่วงของจำนวนสุ่มเป็นอะไร? เปลี่นช่วงตั้งแต่ 0.0 ถึง 1.0 ฟังก์ชั่นเปลี่นประเภท float() และ คงค่า RAND_MAX น่าจะช่วยคุณ


บทที่4.7 มีวิธีคำนวณปริพันธ์จำกัดเขตของฟังก์ชันคณิตศาสตร์มีชื่อจำลอง Monte Carlo โดยใช้ฟังก์ชั่นจำนวนสุ่ม rand() ใน stdlib.h ความคิดเห็นที่สุ่มเพียง:

• สุ่มว่างจุดในขอบเขตของฟังก์ชันคณิตศาสตร์ที่อยากทราบ

• คูณเศษจุดอยู่ใต้ฟังก์ชั่นคณิตศาสตร์กับพื้นที่่่ขอบเขต

• ใช้จุดมากขึ้นความถูกต้งมากขึ้นด้วย

ทำจำลอง Monte Carlo เพื่อคำนวณปริพันธ์จำกัด sin x จาก 0 ถึง π/2 หรือ

   ∫{0→π/2} sin x đx = –cos(π/2) + cos(0) = 1 

และดูความถูกต้องสำหรับ 1000, 5000, 10 000, 50 000, 100 000 จุด นี่ก็เป็นหนึ่งวิธีลองดูคุณภาพฟังก์ชั่นจำนวนสุ่ม


บทที่4.8 มีตัวเลขฐานสิบมีมูลค่าไม่สามารถเก็บไว้ตรงในจำนวนจริง float ช่น 1/5 = 0.2 พยายามทดสอบโดยวนซ้ำนี้และเปรียบเทียบผลกับมูลค่าถูกต้อง

   unsigned short jamNuanSam = 0; /* จำนวนซ้ำ */

   float thangMot = 0.0f;         /* ทั้งหมด */

   while( jamNuanSam < 100000 ) {

     thangMot += 0.2f;

     jamNuanSam++;

   }

ปัญหานี้คล้ายกับปััญหาไม่สามารถเขียน 1/3 โดยตรงใช้กับเลขฐานเศษสิบ
