บทที่ ๓: โครงสร้าง ห้องสมุด ตัวแปร แถวลลำดับ

เพื่อมสามรถเริ่มต้นเขียนโปรแกรมเริ่มต้นเราต้องเข้าใจองค์ประกอบและ 'วัสดุ' จะสร้างมันขึ้นมา โปรแกรมที่ได้สร้างขึ้นจากงบ ฟังก์ชั่น ค่าคง ตัวแปรอื่น ๆ คุณจะเริ่มต้นที่จะเรียนรู้เกี่ยวกับเรื่องดังต่อไปข้างล้างนี้

3.1 โครงสร้างโปรแกรม

แผนภาพด้านล่างแสดงให้เห็นโครงสร้างของโปรแกรมตัวอย่างหนึ่ง โปรแกรมนี้มีฟังก์ชั่นอยู่ในหลายไฟล์แยกกันตามภารกิจของพวกเขา แบ่งโปรแกรมเช่นนี้ช่วยพัฒนามันเร็วขึ้น ต้องคอมไฟล์แฟ้มที่เปลี่ยนแปลงเท่านั้นและผสานเข้าด้วยกัน ทุกโปรแกรม C ต้องมีฟังก์ชั่นที่มีชื่อว่า main() หนึ่งเพียงอย่างเดียว มันปรับคอมไพล์เริ่มวิ่งโปรแกรมจากไหน แต่ไม่จำเป็นฟังก์ชั่น main() ต้องอยู่ในแฟ้มที่มีชื่อ 'main.c'

 

ผู้ที่มีไฟล์ที่สองเรียกว่า:

• 'h' แฟ้มหัว

• 'c' แฟ้มแหล่ง


แฟ้มหัวจะกำหนดพวกฟังก์ชั่นให้ส่วนอื่นๆ ของโปรแกรมได้ใช้มันแต่ไม่ต้องการทราบว่าวิธีทำพวกฟังก์ชั่นนั้นในแฟ้มแหล่ง ดูด้านล่าง:

ภายใน ham.h ระบุพวกฟังก์ชั่น

   void ham0(int tuaPreng0, int tuaPreng1);

   char ham1();


ในแฟ้ม ham.c มีรหัสสำหรับฟังก์ชั่น

   void ham0( int utaPreng0, int tuaPreng1 ) {

      ...

   }


   char ham1() {

      ...

   }


สำหรับโปรแกรมขนาดเล็กที่คุณน่าจะไม่ต้องการเแยกฟังก์ชั่นอยู่ในแฟ้มแตกต่าง ดูตัวอย่างเล็ก main() ที่ด้านล่าง:

  1. #include <stdio.h>  /* ใช้ฟังก์ชั่นฝากคำบนหน้าจอจากห้องสมุด stdio */

  3. /* ฟังก์ชั่นฟากคำสวัสดี */
  4. void fakKhamSawadi() {
  5.   printf ( "สวัสดี\ n");
  6. }

  8. /* ฟังก์ชั่นหัวหน้า */
  9. int main(int argc, ถ่าน ** argv) {
  10.    /* ฟากคำสวัสดี */
  11.    fakKhamSawadi();

  13.    return 1;
  14. }

อีกวิธีหนึ่งคือกำหนดฟังก์ชั่นในแฟ้มก่อน main() และรหัสแหล่งของฟังก์ชั่นมัด fakKhamSawadi() ด้านล่าง:

  1. #include <stdio.h> /* ใช้ฟังก์ชั่นฝากคำบนหน้าจอจากห้องสมุด stdio */

  3. /* กำหนดฟังก์ชั่นฟากคำสวัสดีก่อน */
  4. void fakKhamSawadi();

  6. /* ฟังก์ชั่นหัวหน้า */
  7. int main (int argc, ถ่าน ** argv) {
  8.    /* ฟากคำสวัสดี */
  9.    fakKhamSawadi();

  11.    return 1;
  12. }

  14. /* กำหนดฟังก์ชั่นฟากคำสวัสดี */
  15. void fakKhamSawadi() {
  16.   printf ( "สวัสดี\n");
  17. }

3.2 ห้องสมุดที่นิยมใช้

ในหนังสือเล่มนี้เราจะใช้สี่ห้องสมุดมากที่สุด

stdio.h ห้องสมุดฝากรับมาตรฐาน

stdlib.h ห้องสมุดมาตรฐานห้องสมุด

math.h ห้องสมุดทางคณิตศาสตร์

time.h ห้องสมุดทางเวลา


เวลาใช้พวกห้องสมุดใช้เหล่านี้:

   #include <ชื่อห้องสมุด>


• stdlib.h

มีฟังก์ชั่นมากมีประโยชน์ใน stdlib.h แต่ตอนนี้ฉันเพียงบอกเกี่ยวกับฟังก์ชั่น rand() มันสร้างตัวเลขสุ่มและใช้สำหรับกระบวนการสุ่มเหล่านี้ ก่อนใช้ rand() โดยใช้คำสั่งก่อน:

   srand (time(0));

ในเวลาเริ่มต้นวิ่งโปรแกรม ถ้าไม่เตรียมพร้อมมันจะฝากเรียงตัวเลขสุ่มเหมือนกันทุกครั้งที่คุณวิ่งโปรแกรมของคุณ ฟังก์ชั่น time() อยู่ในห้องสมุด time.h.


• math.h

ห้องสมุด math.h มีฟังก์ชั่นหลายอย่างคือ cos math.h เช่น (), () exp() sin() sqrt() pow() ฯลฯ พวกมันรับแบบ double และฝากจำนวน double ก็มีประเภท float เช่น cosf() tanf() expf() sinf() sqrtf() powf() ฯลฯ พวกฟังก์ชั่นนี้ควรใช้เฉพาะxี่จำเป็นเพราะมันคำนวณช้า ยกตัวอย่างอย่าทำเช่น:

   float gamLangSong = powf(leik, 2.0f);

   float jamNuan0 = 2.0f*cosf(2.0f);

   float jamNuan1 = 4.0f *cosf(2.0f);


ควรทำ:

   float gamLangSong = 2.0f * ลอย 2.0f; /* อย่าใช้ powf() เมื่อไม่จำเป็น */

   float jamNuanCos = leik; /* คำนวณ cosf(2.0f) หนึ่งครั้งเท่ามัน */

   float giaTri0 = 2.0f * giaTriCos;

   float giaTri1 = 4.0f * giaTriCos;

3.3: ข้อมูลประเภทมาตรฐาน

รายการประเภทข้อมูลพื้นฐานภาษา C ตั้งใจช่วงระยของค่าพวกมัน แม้ว่าคอมพิวเตอร์ทันสมัยมีความจำมากแต่คุณควรเลือกประเภทข้อมูเพียงพอใช้กับงานทำ ไม่ใช้แต่จะช่วยประหยัดความจำ จะช่วยเพิ่มความเร็วคำนวณด้วย อ่าน บันทึก คัดลอก ประเภทข้อมูขนาดเล็กได้คำนวณเร็วขึ้นและบางครั้งคำนวณเลขขนานในโปรเซสเซอร์ที่ทันสมัย


จำนวนเต็ม             ความยาว             ช่วงระยะ

                    (ไบต์)

char                   1      –27 → 27 – 1 (–128 → 127)

short                  2      –215 → 215 – 1 (–32768 → 32767)

int                    4      –231 → 231 – 1 (–32768 → 32767)

long                 ≥ 4    ≤ –231 → ≥ 231 – 1 (–2147483648 → 2147483647)

long int               8      –263 → 263 – 1 (−9223372036854775808 → 9223372036854775807)

long long            ≥ 8    ≤ –263 → ≥ 263 – 1 (−9223372036854775808 → 9223372036854775807)

unsigned char          1        0 → 28 – 1 (0 → 255)

unsigned short         2        0 → 216 – 1 (0 → 65535)

unsigned int           4        0 → 232 – 1 (0 → 4294967295)

unsigned long        ≥ 4      ≥ 0 → 232 – 1 (0 → 4294967295)

unsigned long int      8        0 → 264 – 1 (0 → 18446744073709551615)

unsigned long long   ≥ 8      ≥ 0 → 264 – 1 (0 → 18446744073709551615)


จำนาณจริง               ความยาว             ช่วงระยะ  (มราฐาน)

(theo chuẩn IEEE 754)  (ไบต์)

float                  4      ±2–126 → ±(2 – 2–32) × 2127 (≈ ±1,18 × 10–38 → ≈ ±3,40 × 1038)

double                 8      ±2–1022 → ±(2 – 2–64) × 21023 (≈ ±2,23 × 10–308 → ≈ ±1,80 × 10308)

long double            *      ±2–16384 → ±216384 (≈ ±10–4932 → ≈ ±104932)


* แล้วแต่ฮาร์ดแวร์บริ CPU เทลใช้ 80 บิต (10 ไบต์) สำหรับประเภทนี้ แต่คอมไพเลอร์สามารถเอา 12 หรือ 16 ไบต์สำหรับมัน


ประเภท long และ long long สามารถยาวก่วา 4 หรือ 8 ไบต์ขึ้นแล้วแต่ฮาร์ดแวร์และระบบจัดงาน เพื่อสังเกตความยาวประเภทข้อมูลใช้ฟังก์ชั่น sizeof()

ตัวอย่างเช่น:

   sizeof( unsigned char );

   sizeof( float );


ภาษา C คุณสามารถผสมจำนวนเต็มและจริง ใช้ในสตูรการตัวหนึ่ง

   unsigned char ch = 45.6f;  /* กำหนดมีจำนวณจริง */

   float fl = 23;             /* กำหนดมีจำนวณเต็ม */

   double dbl = 123.456;

   printf("ผล: %5.3f\n", (ch - fl)/dbl );


คุณควรระมัดในกรนีใช้ตัวแปรงประเภทไม่เหมื่อนกัน อาจว่าคำนวณเห็ตผูลที่ไม่คาดคิด หากมีการคำนวณจำนวนจริงกับจำนวนเต็ม จำนวนเต็มจะถูกเปลี่ยนเป็นจำนวนจริง แล้วคำนวณเห็ตผูลก็เป็นความจริงด้วย หากว่าคุณต้องการบังคับเปลี่ยนค่าจำนวนเต็มเป็นจริงหรือสลับไป (ค่าจำนวนเต็มเต็มเป็นค่าจำนวนเต็มเต็มอื่น ตัวอย่าง short เป็น char)ใช้ชื่อประเภทก่อนชื่อตัวแปร ค่าคง ฟังก์ชั่นที่คุณอยากเปลี่ยนเช่น:

  (<ชื่อตัวแปร>) <จำนวณ>


ตัวอย่างเช่น:

  (unsigned char) 45.6f    /* = 45  */

  (int) -12.45             /* = -12 */

  (unsigned int) -12.45    /* = 0   */

  int intr = 2341;

  float fl = (float)intr; /* = 2341.0f */


เรียนรู้เพิ่มเติม

หากสามารถเลือกค่าจำนวนเต็มแทนค่าจำนวนจริงทำไปซี ปรกติเลขคณิตจำนวนเต็มเร็วกว่าเลขคณิตจำนวนจริง ฮาร์ดแวร์รุ่นคอมพิวเตอร์ส่วนบุคคลแรกไม่สนับสนุนจำนวนจริง บังคับเขียนซอฟแวร์ทำให้และวิ่งช้ามาก ทันสมัยสบายก่วา คอมพิวเตอร์ส่วนบุคคลมีฮาร์ดแวร์สนับสนุนจำนวนจริง แต่ยังเร็วเท่ากับเลขคณิตจำนวนเต็ม


เลขคณิตบวกเร็วกว่าลบเร็วกว่าคูณเร็วกว่าหาร ซ้ำในกรณีที่จำเป็น บางครั้งคุณสามารถหนีคูณและหารถ้าค่าจำนวนเป็นค่ากำลัง 2 เช่น 2, 4, 8, 16, 32, 64, ฯลฯ ใช้เลื่อนขว่า >> และเลื่อนซาย << วิธีนี้เร็วมาก

   23 * 2 เหมื่อน 23 << 1 (คูณ 2)

   3 * 128 เหมื่อน 3 << 7 (คูณ 128)

   400 / 32 เหมื่อน 400 >> 5 (หาร 32)

จำไว้ว่าคุณไม่ได้สามารถใช้วิธีนี้กับจำนวนจริง

3.4: คำบรรยาย

คำบรรยายสำคัญมากแต่มันไม่มีอิทธิพลอะไรในโปรแกรม มันช่วยคุณจำไว้และเข้าใจรหัสแหล่งคุณได้เขียนหลายๆ เดือนหรือปีมาแล้ว แล้วก็คำบรรยายช่วยนักเขียนโปรแกรมอื่นๆ เข้าใจรหัสแหล่งของคุณ ภาษา C ทันสมัยมีสนับสนุนสองแบบคำบรรยายภาพ:

  1. /* <คำบรรยาย> */

  2. // <คำบรรยาย>


ตัวอย่างเช่น:

   /* คำบรรยายจำกัดในแถวเดียว */

   /* คำบรรยายใช้

        สาม

        แถว */

   // คำบรรยายไปตลอดแถว


วิธีใช้คำบรรยายส่วนตัวมาก สิ่งสำคัญที่สุดคือมันทำให้รหัสเข้าใจง่ายขึ้นสำหรับเพื่อนร่วมงาน


ฉันชอบใช้คำบรรยายแบบ '//' ช่วยหาความผิดพลาดในโปรแกรม มันง่าย'ปิด'ผลกระทบของคำสั่งหรือฟังก์ชั่นกลุ่มหนึ่งโดยคำบรรยายแบบ '/*  */' เพื่อตรวจสอบความผิดพลาดอยู่ที่ไหน

3.5: ประกาศค่าคงและตัวแปร

ภาษา C ทันสมัยให้คุณกำหนดค่าคงและตัวแปรที่ใดก็ได้ รุ่นแรกของภาษา C บังคับกำหนดตัวแปรทั้งหมดก่อนพวกคำสั่ง

<ชนิดข้อมูลของฟังก์ชั่น> <ชื่อฟังก์ชัน> (<เรียงตัวแปรส่งขอมูลเข้าฟังก์ชันถ้ามี>) {

   /* กำหนดตัวแปรทั้งหมด */

    ...


   /* คำสั่ง */

    ...

}

ซึ่งจะทำให้รหัสสับสนมากขึ้น เวลานี้คุณจะใช้จ่ายประกาศสถานที่ที่เหมาะสมของมัน

   <ชนิดข้อมูลของฟังก์ชั่น> <ชื่อฟังก์ชัน> (<ตัวแปรลำดับของฟังก์ชัน ถ้ามี>) {

   /* กำหนดตัวแปร */

    ...


   /* คำสั่ง */

    ...

   /* คำหนดอีกตัวแปร */

    ...


   /* คำสั่ง */

    ...

}


กำหนดตัวค่าคง

มีสองวิธีที่กำหนดค่าคงเช่น:

1. ใช้คำหลัก const:

   const <ประเภทข้อมูล> <ชื่อตัวค่าคง> = <มูลค้า>;

   const float PI = 3.1415926f;  /* π */

   const short NA_KAK = 0x3c00;  /* หน้ากาก */

   const char ok = 'q';          /* ออก */


2. ใช้ #define ตั้งใจว่าไม่ได้กำหนดประเภทข้อมูลและไม่ใช้ ';' สุดท้ายแถว

   #define <ชื่อตัวค่าคง> <มูลค่า>

   #define 3.1415926f PI


ค่าคงไม่สามารถเปลี่ยนแปลงมูลค่าหลังการกำหนด ถ้าคุณพยายามเปลี่ยนมูลค่าของค่าคงคอมไพลร์จะประกาศความผิดพลาด


กำหนดตัวแปร

เมื่อกำหนดตัวแปรคุณก็สามารถกำหนดชื่อตัวแปรไม่มีมูลค่าหรือกำหนดมูลค่าแรก

   <ชนิดข้อมูล> <ชื่อตัวแปร>;          /* ไม่กำหนดค่าแรก */

   <ชนิดข้อมูล> <ชื่อตัวแปร> = <มูลค่า>; /* กำหนดค่าแรก */


ดูตัวอย่าง

   unsigned int khaDanchani; /* ค่าดรรชนี */

   short pi = 2120;          /* ปี */

   char akson = 's';         /* ตัวอักษร */

   unsigned int si = 0x00ff00ff; /* สี */


คุณก็สามารถกำหนดหลายตัวแปรประเภทเดียวกันหากผู้ใช้ ',' ระหว่างมัน:

  float khuamYao, khuamKuang, khuamSung; /* ความยาว ความกวาง ความสูง*/

  float ratMi = 1.0f, khuamReo = 4.0f;   /* รัศมี, ความเร็ว */


ตั้งใจหน่อย: หากคุณพบว่าตัวแปรมีมูลค่าแปลกลองดูยังไม่กำหนดค่าให้มัน


กฎแกงใช้ในการกำหนบ

• ชื่อตัวคงค่าและตัวแปรต้องการเริ่มต้นมีตัวอักษรภาษาละตินหรือตัวสัญลักษณ์ '_' หลังจากนั้นสามารถใช้ภาษาละตินหรือตัวสัญลักษณ์ '_';หรือหมายเลข ดูตัวอย่าง

   float f_fg;

   int __soMoi;

   char r55;

ตัวอย่างไม่ถูกต้อง

   double khuam reo           /* มีช่องว่าง */

   unsigned int 3khrang;      /* เริ่มต้นมีหมายเลข */


• ชื่อค่าคงที่และตัวแปรควรไม่ยาวเกิน 1024 ตัวอักษร ( แต่จำกัดนี้แล้วแต่คอมไพเลอร์)


• ปรกติคนมักจะใช้สองวิธีที่จะเชื่อมต่อสองคนหรือมากกว่าตัวอักษรเมื่อประกาศ

1. ใช้ตัวอักษรใหญ่ให้คำใหม่เช่น (เรียกว่าทรงอูฐ)

   double phernThi;  /* พื้นที่่ */

   unsigned int jamnuanKhonNaiKrungThep; /* จำนวนคนในกรุงเทพฯ */


2. ใช้ตัวสัญลักษณ์ '_' ระหว่างคำใหม่เช่น

   double phern_thi;

   unsigned int jamnuan_khon_nai_krung_thep;


บางคนตำหนิว่าทรงอูฐเพราะเขาคิดมันอ่านยากก่วา มีคนอื่นบ่นว่าใช้ตัวอักษร '_' ต้องกดแป้นพิมพ์ไกลและทรงอูฐสวกก่วา ฉันพยายามทั้งแล้วและชอบทรงอูฐมากก่วา


ชื่อค่าคงนิยมใช้ตัวอักษรใหญ่เช่น:

   const float DOK_BIA = 5.6f;   /* ดอกเบีย */

   const int WINATHI_NAI_NUNG_WAN = 3600;  /* วินาทีในหนึ่งวัน */


• ุกครั้งควรตั้งชื่อชื่อตัวแปรที่จะช่วยให้คุณรู้ว่าเป็นมูลค่าอะไร ตัวอย่าง

   int i, j, k; /* ไม่ทราบว่าตัวแปรนี้เป็นสิ่งอะไร */

ควรเขียน

   int jamNuanKhon, wan, duan; /* จำนวนคน วัน เดือน เข้าใจง่ายก่วา */

 

• ต้องการกำหนดตัวแปรก่อนใช้มัน:

   void function() {

      pi = 2000   /* ผิด ยังไม่กำหนดตัวแปร */

      short pi;   /* กำหนดตัวแปรปี */

      pi = 2020;  /* กำหนดแล้ว ใช้ได้ */

      ...

   }

    

สามารถจำกัดตัวแปรโดย '{' และ '}' ดูด้านล่าง:

   unsigned char munKha0 = 0; /* ประกาศมูลค่า 0 */

   {

       usigned char munKha1 = 5; /* ประกาศมูลค่า 1 */

       printf ( "%d\n", munKha0 + munKha1); /* ฝากค่าบวกสองมูลค่า */

   }

   printf ( "%d\n", munKha0 + munKha1); /* ผิด กำหนกของมูลค่า1หมดแล้ว */


• ในแฟ้มภาษา C คุณสามารถกำหนดตัวแปรเรียกว่าตัวแปรทั่วโลก ตัวแปรแปรทั่วโลกที่ใช้ในฟังก์ชั่นทั้งหมดในแฟ้มนั้น ตัวแปรกำหนดในฟังก์ชั่นเรียกว่าตัวแปรท้องถิ่น มันถูกจำกัดใช้เฉพาะในฟังก์ชั่นนั้น

  float khanat = 0.0f;           /* ขนาด ตัวแปรทั่วโลก */


   <ฟังก์ชั่น 0> {

       unsigned char kiKrang = 1; /* กี่ครั้ง ตัวแปรท้องถิ่น */

       khanat = 3.0f;             /* ตั้งตัวแปรทั่วโลกขนาด = 3.0f */

       ...

   }


   <ฟังก์ชั่น 1> {

      unsigned char kiKhon = 0;   /* กี่คน ตัวแปรท้องถิ่น */

      khanat = 5.0f;              /* ตั้งตัวแปรทั่วโลกขนาด = 5.0f */

      ...

   }

   ...


ปรกติคุณควรไม่ใช้ตัวแปรทั้วโลกเพราะเป็นเรื่องยากจะทราบว่าฟังก์ชั่นตัวใดจะเปลี่ยนแปลงมูลค่าของตัวแปร ฟังก์ชั่นใดก็สามารถเปลี่ยนมูลค่าของตัวแปรทั่วโลกทำตรวจสอบความผิดพลาดอู่ด้านไหน กรนีหนึ่งไม่ตามกดนี้คือการเขียนโปรแกรมเชิงวัตถุและตัวแปรทั่วโลกเป็นตัวแปรของวิชากำหนด ตัวอย่างภาษาวัตถุ: C++ ObjectiveC Java Swift


• ภาษา C สังเกตตั้วอักษรเล็กและใหญ่  ทุกตัวแปรข้างล้งนี้เป็นตัวแปรแตกต่างกัน:

   float so;

   float SO;

   float sO;

   float So;

คุณไม่ควรตั้งชื่อตัวแปรคล่ายกัน ทำแยกแยะตัวแปรยาก


• คุณสามารถกำหนบตูลค่าโดยสามเลขฐานที่ภาษา C สนับสนุนหรือใช้มูลค่าของตัวอักษรตัวสัญลักษณ์:

   unsigned short giaTriThap = 0x00ff; /* เลขฐานสิบหก */

   int jamNuan = 070745;               /* เลขฐานแปด */

   unsigned char nakRian = 123;        /* เลขฐานสิบ */

   float sanyalak = '@';               /* ตัวสัญลักษณ์ */


ระวัง: อย่ากำหนดตัวแปรแทนตัวแปรอื่นในฟังก์ชัน ตัวอย่างด้านล่างเป็นฟังก์ชั่นบวกขึ้นมูลค่าพึ้นทีกับพื้นที่่่ที่มีอยู่แล้ว แต่ว่าไม่ระมดระวังและตั้งตัวแปรระหว่าง '{' และ '}'ที่มีชื่อ tarang (ตาราง) เดียวกับพารามิเตอร์ของฟังก์ชั่น ตัวแปรใหม่จะแทนพารามิเตอร์ของฟังก์ชั่น

   float khamNuanTarang( float khuamKuang, float khuamSung, float tarang ) {

      ...

      {

         float tarang = khuamKuang * khuamSung;

      }

      return tarang;

   }


เห็ตผลของฟังก์ชั่นนี้คือส่งขึ้นมูลค่าพารามิเตอร์ tarang ไม่บวกเพิ่มมูลค่า khuamKuang * khuamSung แสดงเฉพาะพื้นที่่เก่า ฟังก์ชั่นนี้เป็นอย่างง่ายและตรวจสอบหาเจอความผิดพลาดนี้ไม่ยาก แต่ในฟังก์ชั่นสับสนหาเจอความผิดพลาดอย่างนี้อาจไม่ง่ายจริง (ฉันเคยได้แล้ว) แต่คอมไพเลอร์จะไม่หาเจอตรวจสอบมัน! ระมัดระวังในเวลากำหนดชื่อตัวแปรไม่เหมื่อนพารามิเตอร์!

3.6 ตัวแปรชุดและสายอักขระ

มีกรณีหที่ต้องการมีช่วงมูลค่าเช่นข้อมูลอุณหภูมิสำหรับปีที่ผ่านมา รายการของศัตรูในเกม พิกเซลสำหรับรูป เรียงอักษรของประโยชน์ที่อยากฝากบนหน้าจอ ฯลฯ กำหนดตัวแปรหนึ่งสำหรับทุกชิ้นข้อมูลไม่สะดวก รูปภาพหนึ่งสามารถมีหลายล้านพิกเซล เราต้องมีช่วงข้อมูลที่เรียกว่าตัวแปรชุด ตัวแปรชุดได้สร้างขึ้นในความจำเป็นชุดหน่วยหนึ่ง


บางรูปแบบกำหนดตัวแปรชุดหนึ่งมิติ

   <ชนิดข้อมูล> <ชื่อตัวแปรชุด>[<ความยาว>];

   <ชนิดข้อมูล> <ชื่อตัวแปรชุด>[<ความยาว>] = {<มูลค่า 0> <มูลค่า 1> ... };

   <ชนิดข้อมูล> <ชื่อตัวแปรชุด>[] = {<มูลค่า 0> <มูลค่า 1> ... };


ความยาวประกอบจำนวนสมาชิกในตัวแปรชุด ไม่ใช้จำนวนไบต์ของตัวแปรชุด 


คุณสามารถตั้งมูลค่าเมื่อกำหนดตัวแปรชุด ล้อมรอบพวกมูลค่าโดย '{' และ '}' และใช้ ',' ระหว่างแต่ละมูลค่า จำนวนตัวมูลค่าไม่สามรถมากกว่าความยาวตัวแปรชุด (น้อยก่วาไม่เป็ยไร) แต่คุณสามารถไม่กำหนดจำนวนสมชิดและปล่อยให้คอมไพเลอร์นับจำนวนตัวมูลค่า


ดูตัวอย่างบางด้านล่าง:

   float khuamRon[365]; /* ตัวแปรชุดความร้อนเป็น float มี 365 สมาชิก */

   unsigned char khuamSung[5] = {155, 148, 175, 166, 170}; /* ตัวแปรชุดความสูงเป็น uint มี 5 สมาชิก */

   char chu[] = { 'T', 'h', 'a' 'k', 's', 'i', 'n', 0};    /* ชื่อ กำหนดมูลค่าโดยตัวสัญลักษณ */


ในภาษา C ตัวแปรชุดที่มีความยาว n สมาชิก สมาชิกเริ่มต้นมีดัชนี 0 และสมาชิกสุดท้ายคือ n - 1. ในตัวแปรชุดความสูงจากจากข้างบน สมาชิก 0 และ 4 มีมูลค่า:

   khuamSung[0] = 155;

   khuamSung[4] = 170;


ระวัง 1: ระมัดระวังไปนอกตัวแปรชุด (ดัชนีสูงก่วา n - 1) คุณไม่สารารถทราบมีสิ่งที่มีอยู่ อันตรายมากก่วาคือเก็บไว้มูลค่าที่โื้น คุณสามารถบันทึกบนคำสั่งโปรแกรม ทำโปรแกรมทำงานผิด แฮกเกอร์รักใช้ประโยชน์นี้ในโปรแกรมไม่ตรวจสอบถ้กำหลังเก็บไว้มูลค่าภายในหรือนอกตัวแปรชุด 


ระวัง 2: อย่าลืมกำหนดมูลค่าตัวแปรชุดก่อนเอาไปช้  เมื่อคอมไพลร์เอาความจำเพื่อสร้างตัวแปรชุดมันไม่กำหนดมูลค่า ไม่สามารถทราบว่ามีมูลค่าอะไรบ้าง


พวกตัวแปรชุดที่เราเห็นแล้วข้างบนเป็นตัวแปรชุดหนึ่งมิติ ก็มีดัวแปรชุดหลายมิติด้วย:

   <ชนิดข้อมูล> <ชื่อตัวแปรชุด> [<ความยาว 0>] [<ความยาว 1>] ...

   <ชนิดข้อมูล> <ชื่อตัวแปรชุด> [] = {{<มูลค่า 00>, <มูลค่า 01> ... }, {<มูลค่า 10>, <มูลค่า 11> ... }};


ตัวอย่างเช่น:

   unsigned char leik[5][2] = {{0xff, 0x05, 0x34, 0x4e, 0x1f}, {0x00, 0x47, 0xa0, 0x20, 0x8f}}; /* เลข */

   float phiKat[256][256][256]; /* พิกัด, ตัวเรียงสามมิติ */


ปรกติตัวแปรชุดได้สร้างขึ้โดยหนึ่งหน่วยความจำต่อเนื่อง ไม่เกรยวเ)็นหนี่งหรือหลายมิติ คุณสามารถสร้างตัวแปรชุดหลายมิติโดยตัวแปรชุดหนึ่งมิติ ตัวอย่างเช่นมีรูปภาพขนาด k × s พิกเซล (ความกว้างและความสูง) คุณสามารถกำหนดำตัวแปรชุดมีข้อมูลเช่น:

   int rupPhap[r][c]; /* ตัวแปรชุดสองมิติ */

หรือ

   int rupPhap[r*c]; /* ตัวแปรชุดหนึ่งมิติ */

ทันทีที่คุณน่าจะคิดว่าใช้วิธีไหนได้ติดต่อพิกเซล คอลัมน์ x และแถว y ในภาพตัวแปรชุดหนึ่งแถว หากภาพได้จัดระเบียบแถว (ส่วนใหญ่รูปแบบภาพเช่น JPEG, PNG, EXR จัดระเบียบแถว), คูณความกว้างกับจำนวนแถวและบวกหมายเลขคอลัมน์ ให้ดูรหัสข้างล้าง:

   unsigned int pixel = [x + r*y];   /* พิกเซลอยู่ที่คอลัมน์ x แถว y */


• สายอักขระเป็นตัวแปรชุดที่มีประเภท char และได้ยกเลิกโดยตูลค่า '0x00' เรียกว่าตัวสัญลักษณ์ศูนย์ แต่ไม่มีข้อมูลเกี่ยวกับความยาว (สายอักขระของภาษาอื่นๆ อ่าจเก็บไว้ข้อมูลความยาว) แต่ความยาวถูกจำกัดโดยความจำและสถาปัตยกรรมคอมพิวเตอร์ที่กำลังใช้ยู่ ตัวสัญลักษณ์ศูนย์สำคัญมากเพราะมันแสดงให้ทราบว่าสายอักขระจบที่ไหน:

   char <ชื่อสายอักขระ> = "<ตัวเรียงอักษร/ตัวสัญลักษณ์>";


เมื่อกำหนดสายอักขระไม่จำเป็นตั้งใจตัวสัญลักษณ์ศูนย์มาก คอมไพเลอร์จะเพิ่มมันให้ แต่ถ้าคุณกำหนดสายอักขระเช่นนี้คุณต้องการว่างสุดถ้ายเอ็ง:

   char sawadi[] = "Sawdi";                         /* สวัสดี ไม่ต้องการว่าง */

   char sawadi[] = { 'S', 'a', 'w ', 'd', 'i', 0};  /* สวัสดี ต้องการว่างตัวสัญลักษณ์ศูนย์สุดถ้าย */


คุณสามารถรับสายอักขระอย่างนี้:

   #include "stdio.h" /* ห้งสมุดรับฝาก */

   ...

   char namSakun[255]; /* นามสกุล */

   unsigned char ayu;  /* อายุ */


   /* ข้อมูล */

   printf ( "กรุณาพิมพ์นามสกุล:");

   scanf ( "%s" namSakun );

   printf ( "กรุณาพิมพ์อายุ:");

   scanf ( "%d", &ayu );


   /* พิมพ์บนหน้าจอ */

   printf ( "นามสกุล: %s อายุ: %d\n" namSakun, ayu );


เรียนรู้เพิ่มเติม

สำหรับสายอักขระโดยใช้ตัวอักษรเช่น "#dr^6" ตัวอักษรละรหัสเลข แต่ถ้าคุณมีตัวอักษรเช่น "อีดиすណ" ไม่ได้ตัวอักษรละรหัสเลข บางมีถึง 2-4 รหัสเลขตอ่ตัวอักษร! ระบบรหัสเลข UTF8 ดูรหัสสำหรับตัวอักษรของสองสายอักขระด้านล่าง:

"dr^6" → 0x23 0x64 0x52 0x36 0x5e

"อีดиすណ" → 0xe1 0xbb 0xd0 0x94 0x81 0xb8 0xe0 0xe3 0xb8 0x99 0x81 0x8e 09e 0xe1

3.7 ฟังก์ชัน scanf ()

มีฟังก์ชั่น printf() เพื่อฝากข้อมูลบนหน้าจอและมีฟังก์ชั่น scanf() (ฟังก์ชั่นสแกน) สำหรับรับข้อมูล หลังจากได้เรียนรู้เกี่ยวกับตัวแปรฉันสามารถอธิบายวิธีการใช้ฟังก์ชั่นนี้ มันใช้หัสควบคุมเช่นเดียวกับ printf() และช่วยให้แป่ลสายอักขระรับ มันก็อยู่ในห้องสมุด <stdio.h> ดูอย่างสำหรับมัน

   scanf ( "<อักขระควบคุม>", <ที่อยู่ตัวแปร>);


เพื่อรับข้อมูลใช้ตัวสัญลักษณ์ '&' ก่อนชื่อตัวแปร มันหมายความว่าที่อยู่ของตัวแปร ฉันจะอธิบายเพิ่มเติมเกี่ยวกับเรืองในบทที่ 6 เวลาเรียนเกี่ยวกับตัวชี้ ฉันแนะนำฟังก์ชั่น scan() เพื่อช่วยคุณสามารถเขียนโปรแกรมได้รับข้อมูล ฟังก์ชั่น scanf() ต้องมีอย่างน้อยหนึ่งตัวแปรสำหรับรับข้อมูล ตัวอย่าง

   unsigned char leik0; /* หมายเลข 0 */

   unsigned char leik1; /* หมายเลข 1 */


   /* รับข้อมูล */

   printf ( "กรุณาพิมพ์ตัวเลข 0:");

   scanf ( "% s", &leik0 );

   printf ( "กรุณาพิมพ์หมายเลข 1:");

   scanf ( "% d", &leik1 );


   /* ฝากบนหน้าจอ */

   printf ( "% d x% d =% d \ n", so0, number1, so0 * ครั้งที่ 1);


จำไว้ว่ากดปุ่ม Enter หรือปุ่ม Return เมื่อพิมพ์เสร็จแล้ว คุณสามารถพิมพ์ข้อมูลทั้งหมดใช้แถวหนึ่ง:

   unsigned char leik0;   /* หมายเลข 0 */

   unsigned char leik1;   /* หมายเลข 1 */


   /* ข้อมูล */

   printf( "กรุณาพิมพ์หมายเลข 0 และ 1 (ใช้ช่องว่างระหว่างหมายเลข):");

   scanf( "%s %d", &leik0 &leik1);


   / * ใส่ขึ้นหน้าจอ * / การ

   printf ( "% d x% d =% d \ n", so0, number1, so0 * ครั้งที่ 1);

ฉันออกจากช่องว่างระหว่างสองรหัสตัวละครแต่ละตัว

   unsigned short jamnuanPhuYing = 45;  /* จำนวนของผู้หญิง  */

   unsigned short jamnuanPhuChai = 56;  /* จำนวนคนของผู้ชาย */

   unsigned short ruam = jamnuanPhuYing + jamnuanPhuChai; /* รวม */

   float suanPhuYing = jamnuanPhuYing / ruam;    /* ส่วนผู้หญิง */

   float suanSoPhuChai = jamnuanPhuChai / ruam;  /* ส่วนผู้ชาย */

   printf( "ผู้หญิงคะแนน: %5.3f\n");

• คะแนน

• จำนวนชีวิด

• หมายเลขปึนกำลังใช้อยู่

• ระยะห่างจากโลก
